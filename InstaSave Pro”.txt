const Downloader = ({ showToast }) => {
  const [url, setUrl] = useState('');
  const [status, setStatus] = useState('idle'); // idle, loading, success, error
  const [errorMsg, setErrorMsg] = useState('');
  const [mediaData, setMediaData] = useState(null);

  const handleProcess = () => {
    if (!url.trim()) {
      setErrorMsg('Please enter a valid URL');
      setStatus('error');
      return;
    }
    if (!url.includes('instagram.com')) {
      setErrorMsg('Invalid Instagram URL. Please check the link.');
      setStatus('error');
      return;
    }

    setStatus('loading');
    setErrorMsg('');
    
    // Simulate API Call
    setTimeout(() => {
      setStatus('success');
      setMediaData({
        thumbnail: 'https://images.unsplash.com/photo-1611162617474-5b21e879e113?ixlib=rb-4.0.3&auto=format&fit=crop&w=600&q=80',
        author: '@creative_user',
        desc: 'Beautiful moments captured today! ✨📸 #photography',
        type: url.includes('reel') ? 'video' : 'image',
        likes: '12.4k'
      });
      showToast('Media ready for download!');
    }, 2000);
  };

  const executeDownload = (type) => {
    showToast(`Downloading ${type}...`);
    // Simulated download process
    setTimeout(() => {
      showToast('Download complete!');
      setUrl('');
      setStatus('idle');
    }, 1500);
  };

  return (
    <div className="w-full max-w-3xl mx-auto relative z-10">
      {/* Input Area */}
      <div className={`bg-slate-800/80 backdrop-blur-xl p-2 rounded-2xl border transition-all duration-300 ${status === 'error' ? 'border-red-500/50 shadow-[0_0_20px_rgba(239,68,68,0.2)]' : 'border-white/10 shadow-2xl'}`}>
        <div className="flex flex-col sm:flex-row gap-2">
          <div className="relative flex-1">
            <div className="absolute inset-y-0 left-4 flex items-center pointer-events-none">
              <Link2 className="text-gray-400" size={20} />
            </div>
            <input
              type="text"
              value={url}
              onChange={(e) => {
                setUrl(e.target.value);
                if(status === 'error') setStatus('idle');
              }}
              placeholder="Paste Instagram URL here..."
              className="w-full bg-slate-900/50 text-white placeholder-gray-500 rounded-xl py-4 pl-12 pr-4 focus:outline-none focus:ring-2 focus:ring-purple-500/50 transition-all"
            />
          </div>
          <Button onClick={handleProcess} disabled={status === 'loading'} className="sm:w-auto w-full py-4">
            {status === 'loading' ? (
              <span className="flex items-center gap-2">
                <svg className="animate-spin h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
                  <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
                  <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
                </svg>
                Processing
              </span>
            ) : 'Get Download'}
          </Button>
        </div>
      </div>

      {/* Error Message */}
      {status === 'error' && (
        <div className="mt-3 text-red-400 text-sm font-medium px-4 text-center animate-pulse">
          {errorMsg}
        </div>
      )}

      {/* Result Card */}
      {status === 'success' && mediaData && (
        <div className="mt-6 bg-slate-800/80 backdrop-blur-xl border border-white/10 rounded-2xl p-4 sm:p-6 shadow-2xl animate-in fade-in slide-in-from-bottom-4 duration-500">

              </div>
            </div>
            
            {/* Info & Actions */}
            <div className="w-full sm:w-2/3 flex flex-col justify-between">
              <div>
